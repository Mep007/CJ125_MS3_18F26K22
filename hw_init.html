void HW_Init() {
   // PORT A definition
   PORTA = LATA = 0;
   ANSELA = 0b00100111;         // RA0-2,5 => A/D, RA3 - I/O IN (L_BUT0)
   TRISA  = 0b00101111;         // RA4- CCP5 - DAC2 PWM out

   // PORT B definition
   PORTB = LATB = 0;
   ANSELB = 0;                // all digitals I/O
   TRISB = 0b10000010;        // RB0- PWM CCP4 OUT spojen cinem s RB1(INPUT) - PIC_HEATER (pac RB1 nema CCPx unit)

   // PORT C definition
   PORTC = LATC = 0b00000011;   // MAX7219 nCS OFF :> RC0 = log1, Bosch CJ125 - RC1 = log1 (OFF), RC2 - Bosch CJ125 - RST (log1 = working)
   TRISC =        0b10010000;
   ANSELC = 0;                // all digitals I/O

   //CCP4 PWM
   PWM4_Init(1000);
   PWM4_Set_Duty(0);
  // PWM4_Start();

   OSCCON = 0b01110010;         // 16Mhz int osc
   while (!HFIOFS_bit);


  // UART 1 115k2 Init - oporti UART1_Init usetri 70 bytes
   TXSTA1 = RCSTA1 = 0;
   BRG16_BAUD1CON_bit = 1;    // high speed UART
   SPBRGH1 = 0; SPBRG1 = 34;  // for 16Mhz @115k2 (0.79% error)
   BRGH_TX1STA_bit = 1;
   SPEN_RC1STA_bit = TXEN_TX1STA_bit = CREN_RC1STA_bit = 1;  // povoleni Tx i Rx, zpnuti UART1
   RC1IF_bit = 0;
   RC1IE_bit = PEIE_GIEL_bit = 1;

  //  Init functions to run
   MAX7219_CS = CJ125_CS = 1;       // SPI devices DISABLED
   MAX7219_Init(_MAX7219_SHOW_DIGITS_0_3, _MAX7219_INTENSITY_2);
   MAX7219_DisplayOnOff(0);
   CJ125_RST = 0;                   // 1 => CJ125 ENABLED
  // CJ125_RST = 1;                   // 1 => CJ125 ENABLED //presunutu do TEST fce
   // SPI1_Init_Advanced(_SPI_MASTER_OSC_DIV16, _SPI_DATA_SAMPLE_END, _SPI_CLK_IDLE_LOW, _SPI_LOW_2_HIGH);   // OK pro MAX7219  - viz prislusne .C
   // SPI1_Init_Advanced(_SPI_MASTER_OSC_DIV64, _SPI_DATA_SAMPLE_END, _SPI_CLK_IDLE_LOW, _SPI_HIGH_2_LOW);  // OK pro CJ125  - viz prislusne .C
   ADC_Init();
   LED_Welcome();  // sekvence LED1,2,3 pri zapnuti

  
   GIE_GIEH_bit = 1 ; // Global IRQ ENABLE
}
